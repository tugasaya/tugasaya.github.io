Jawaban:

a. 2x² + x - 10 = 0

2x²-4x+5x-10 = 0

(2x²-4x) + (5x-10) = 0

2x(x-2) + 5(x-2) = 0

(2x+5)(x-2) = 0

2x = -5 atau x = 2

x = -5/2

HP = {-5/2, 2}

b. 3x²-7x-6 = 0

3x²+2x-9x-6 = 0

(3x²+2x) + (-9x-6) = 0

x(3x+2) - 3(3x+2) = 0

(x-3)(3x+2) = 0

x = 3 atau 3x = -2

x = -2/3

HP = {-2/3, 3}

c. 4x²+4x-3 = 0

4x²-2x+6x-3 = 0

(4x²-2x) + (6x-3) = 0

2x(2x-1) + 3(2x-1) = 0

(2x+3)(2x-1) = 0

2x = -3 atau 2x = 1

x = -3/2 x = 1/2

HP = {-3/2, 1/2}
