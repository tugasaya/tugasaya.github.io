[tex] \cot(60) = \frac{1}{ \sqrt{3} } \\ perbandingan \: = > \frac{1}{ \sqrt{3} } = \frac{40}{x} \\x = 40\sqrt{3} [/tex]

[tex] \csc(60) = 2 = \frac{2}{1} \\ perbandingan => \frac{2}{1} = \frac{x }{40 \sqrt{3} } \\ x = 80 \sqrt{3} [/tex]
