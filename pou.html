2a + a - 6 = 90

3a = 90 + 6

3a = 96

a = 96 : 3

<POQ = <SOT

= (2a - 6)°

= (64 - 6)°

<POQ + <POU + <TOU = 180°

<POU = 180° - <POQ - <TOU

<POU = 122° - 46°

<POU = 76°
